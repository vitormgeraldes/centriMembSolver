	\begin{multicols}{2}
		
		\section*{Overview}
		The O--CMF cartridge converts rotor speed into high transmembrane pressure
		\emph{without} external pumps. The orthogonal membrane orientation
		minimizes concentration polarization and preserves membrane performance.
		
		\textbf{Applications:} Clarification, buffer exchange, protein concentration
		
		\medskip
		
		\textbf{Rotor Compatibility:}
		\begin{itemize}
			\item Beckman Coulter JA--20 swinging-bucket rotor
			\item $8 \times 50$ mL tube positions
		\end{itemize}
		
		\columnbreak
		
		\section*{Operating Range}
		\begin{itemize}
			\item \SIrange{11}{13}{\milli\litre} feed volume
			\item \SIrange{2}{60}{bar} equivalent \(\Delta p\) (nominal)
			\item \SIrange{3000}{20000}{RPM} depending on volume and target pressure
		\end{itemize}

	\section*{Centrifugal Pressure Equation}
	
	\[
	\boxed{
		\Delta p_\text{bar}(V,N) =
		5.49\times10^{-6}\;
		N^2\;
		\Big[
		0.0762^2 -
		(0.0713 - 0.0015V)^2
		\Big]
	}
	\]
	
	\subsection*{RPM Requirement (Inverse Relation)}
	\[
	\boxed{
		N(V,\Delta p_\text{bar}) =
		\sqrt{
			\frac{\Delta p_\text{bar}}
			{5.49\times10^{-6}\;
				\big[0.0762^2 -
				(0.0713 - 0.0015V)^2\big]}
		}
	}
	\]
